% Screenshot with caption: \screenshot{path}{caption}
% Renders at up to full textwidth with a minimum rendered width (0.55\textwidth)
% so narrow crops (header, footer, login card) stay legible rather than
% shrinking to postage-stamp size. Height is capped so a tall crop still
% leaves room for the caption.
\newcommand{\screenshot}[2]{%
  \begin{figure}[H]
    \centering
    \IfFileExists{screenshots/#1}{%
      \includegraphics[width=\textwidth,height=0.75\textheight,keepaspectratio]{screenshots/#1}%
    }{%
      \fbox{%
        \parbox{0.9\textwidth}{%
          \vspace{3em}
          \centering\textcolor{appGray}{\textit{Screenshot pending}}\\[0.5em]
          \texttt{\small \detokenize{#1}}
          \vspace{3em}
        }%
      }%
    }
    \caption{#2}
  \end{figure}
}

% Narrow / region-crop screenshot: \screenshotnarrow{path}{caption}
% For small sub-UI captures (login card, modal, user menu dropdown, etc.)
% that look small at full-textwidth scale. Renders at 0.65\textwidth so
% content is easy to read without stretching a small source image.
\newcommand{\screenshotnarrow}[2]{%
  \begin{figure}[H]
    \centering
    \IfFileExists{screenshots/#1}{%
      \includegraphics[width=0.65\textwidth,height=0.75\textheight,keepaspectratio]{screenshots/#1}%
    }{%
      \fbox{%
        \parbox{0.6\textwidth}{%
          \vspace{2em}
          \centering\textcolor{appGray}{\textit{Screenshot pending}}\\[0.5em]
          \texttt{\small \detokenize{#1}}
          \vspace{2em}
        }%
      }%
    }
    \caption{#2}
  \end{figure}
}

% Wide / thin-strip screenshot: \screenshotstrip{path}{caption}
% For very wide narrow captures (header bar, status footer). Caps HEIGHT at
% 0.15\textheight rather than fitting to width, so the image doesn't blow up
% vertically into a skinny unreadable strip.
\newcommand{\screenshotstrip}[2]{%
  \begin{figure}[H]
    \centering
    \IfFileExists{screenshots/#1}{%
      \includegraphics[width=\textwidth,height=0.12\textheight,keepaspectratio]{screenshots/#1}%
    }{%
      \fbox{%
        \parbox{0.9\textwidth}{%
          \vspace{1em}
          \centering\textcolor{appGray}{\textit{Screenshot pending}}\\[0.3em]
          \texttt{\small \detokenize{#1}}
          \vspace{1em}
        }%
      }%
    }
    \caption{#2}
  \end{figure}
}

% Two side-by-side screenshots
\newcommand{\screenshotpair}[4]{%
  \begin{figure}[H]
    \centering
    \begin{subfigure}[t]{0.48\textwidth}
      \centering
      \IfFileExists{screenshots/#1}{%
        \includegraphics[width=\textwidth,height=0.75\textheight,keepaspectratio]{screenshots/#1}%
      }{%
        \fbox{%
          \parbox{0.85\textwidth}{%
            \vspace{2.5em}\centering\textcolor{appGray}{\textit{Screenshot pending}}\\[0.3em]
            \texttt{\scriptsize \detokenize{#1}}\vspace{2.5em}%
          }%
        }%
      }
      \caption{#2}
    \end{subfigure}
    \hfill
    \begin{subfigure}[t]{0.48\textwidth}
      \centering
      \IfFileExists{screenshots/#3}{%
        \includegraphics[width=\textwidth,height=0.75\textheight,keepaspectratio]{screenshots/#3}%
      }{%

        }%
      }
      \caption{#4}

  \end{figure}
}
